locais (
  id PRIMARY KEY,
  nome VARCHAR(100) NOT NULL,            -- ex: "Laboratório Particular", "Auditório"
  tipo VARCHAR(50) NOT NULL,             -- ex: "bloco", "setor", "sala", "recepcao"
  pai_id INTEGER REFERENCES locais(id),  -- ex: "tipo: setor, pai_id: 1; locais where id == 1 = bloco A"
  andar VARCHAR(30),                     -- ex: "Térreo", "1º Andar", "Subsolo"
  coordenada VARCHAR(50),                -- ex: "C-T-04" → código interno para mapas
  descricao TEXT,
  ativo BOOLEAN DEFAULT true
);


referencias_locais (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  local_id INTEGER NOT NULL REFERENCES locais(id) ON DELETE CASCADE,
  imagen VARCHAR(255) NOT NULL                    -- ex: "URL da imagen"
  descricao VARCHAR(255) NOT NULL                 -- ex: "Uma descricao para referencia, ou a propria referencia em texto"
);

exames (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  nome VARCHAR(150) NOT NULL,             -- ex: "hemograma", "mamografia"
  descricao TEXT,
  local_id INTEGER NOT NULL REFERENCES locais(id),
);

eventos (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  titulo VARCHAR(150) NOT NULL,
  descricao TEXT,
  data_inicio TIMESTAMP,
  data_fim TIMESTAMP,
  local_id INTEGER NOT NULL REFERENCES locais(id),
  publico_alvo VARCHAR(100)
);

noticias (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  titulo VARCHAR(200) NOT NULL,
  resumo TEXT NOT NULL,
  conteudo TEXT NOT NULL,
  data_publicacao TIMESTAMP,
  local_id INTEGER REFERENCES locais(id),        -- opcional (NULL se noticia não tiver um local)
  tags TEXT[]                                    -- ex: {"inauguração", "abril rosa"}
);

vagas (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  cargo VARCHAR(150) NOT NULL,
  resumo TEXT NOT NULL,
  descricao TEXT,
  requisitos TEXT[],
  data_publicacao TIMESTAMP,
  data_encerramento TIMESTAMP,
  status VARCHAR(20) NOT NULL DEFAULT 'ativa',   -- "ativa", "encerrada"
  local_id INTEGER NOT NULL REFERENCES locais(id),
  como_se_inscrever TEXT NOT NULL,
  tipo_vinculo VARCHAR(50)                       -- ex: "CLT", "Estágio"
);


informacoes_institucionais (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  tipo VARCHAR(50) NOT NULL UNIQUE,              -- ex: "sobre", "lgpd", "missao"
  conteudo TEXT NOT NULL
);

informacoes_gerais (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  chave VARCHAR(50) NOT NULL UNIQUE,             -- ex: "horario_recepcao"
  valor TEXT NOT NULL                            -- ex: "Seg-Sex - 7:20/17:00"
);
